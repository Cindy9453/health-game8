<script>
let heroHP = 100;
let monsterHP = 100;
let used = [];

const questions = [
{q:"每天建議睡眠幾小時？",a:["3","5","7-9","12"],c:2},
{q:"哪個是健康飲食？",a:["炸雞","蔬菜","糖果","汽水"],c:1},
{q:"BMI過高代表？",a:["過瘦","正常","過重","身高高"],c:2},
{q:"每天建議喝水量約？",a:["500ml","1000ml","2000ml","5000ml"],c:2},
{q:"哪個運動最好？",a:["不動","規律運動","熬夜","暴飲暴食"],c:1}
];

// 自動補到100題
for(let i=questions.length;i<100;i++){
questions.push({q:"健康問題 "+(i+1),a:["A","B","C","D"],c:Math.floor(Math.random()*4)});
}

function getQuestion(){
if(used.length===100){alert("通關！");location.reload();}
let idx;
do{idx=Math.floor(Math.random()*questions.length);}while(used.includes(idx));
used.push(idx);
return questions[idx];
}

function updateHP(){
document.querySelector("#heroHP .hp").style.width = heroHP+"%";
document.querySelector("#monsterHP .hp").style.width = monsterHP+"%";
}

function loadQ(){
const q = getQuestion();
document.getElementById("question").innerText = q.q;
const choicesDiv = document.getElementById("choices");
choicesDiv.innerHTML = "";
q.a.forEach((choice,i)=>{
const btn=document.createElement("button");
btn.innerText=choice;
btn.onclick=()=>{
if(i===q.c){
monsterHP-=10;
document.getElementById("hit").play();
}else{
heroHP-=10;
document.getElementById("hurt").play();
}
updateHP();
checkGame();
loadQ();
};
choicesDiv.appendChild(btn);
});
}

function checkGame(){
if(heroHP<=0){alert("你輸了");location.reload();}
if(monsterHP<=0){alert("你贏了");location.reload();}
}

window.onload=()=>{
document.getElementById("bgm").play();
loadQ();
}
</script>
